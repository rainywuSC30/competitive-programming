## 算法1
##### 二分图最大匹配

1. 行和列1对1匹配，使用匈牙利算法求最大匹配
2. match[i] 表示第i列匹配的行 ，除此之外，建立rematch数组，rematch[i]表示第i行匹配的列
3.  rematch[match[i]] = i;
4.  遍历每列 i，对于 match[i] != i 的行，将 match[i]行 和 match[rematch[i]] 交换
   每次更新数组，rematch[match[i]] = rematch[i], match[rematch[i]] = match[i];

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
#define x first
#define y second
typedef long long LL;
typedef pair<int, int> PII;

const int N = 110, M = 4e5 + 10, INF = 0x3f3f3f3f;

int n;
int g[N][N];
int match[N]/*表示i列匹配的行*/, rematch[N]/*表示i行匹配的列*/;
bool st[N];

bool find(int u){
    for (int v = 1; v <= n; v++) {
        if (!st[v] && g[u][v]) {
            st[v] = true;
            if (!match[v] || find(match[v])){
                match[v] = u;
                return true;
            }
        }
    }
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    while(cin >> n) {
        memset(match, 0, sizeof match);
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                cin >> g[i][j];
            }
        }
        int res = 0;
        for (int i = 1; i <= n; i++) {
            memset(st, false, sizeof st);
            if (find(i)) res++;
        }
        if (res != n) {
            cout << -1 << endl;
            continue;
        }
        vector<PII> ans;
        // i 列匹配的行匹配的列是 i
        for (int i = 1; i <= n; i++) rematch[match[i]] = i;
        for (int i = 1; i <= n; i++) {
            if (i == match[i]) continue;
            else {
                ans.push_back({match[i], i});
                match[rematch[i]] = match[i];
                rematch[match[i]] = rematch[i];
            }
        }
        cout << ans.size() << endl;
        for (auto [a, b]: ans) cout <<  "R " <<  a << " " << b << endl;
    }

    return 0;
}
```
