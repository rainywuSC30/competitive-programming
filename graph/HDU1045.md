## 算法1
##### 网络流

对于同行或同列的墙两边的区域标号不同，然后左部为行块，右部为列块，连边容量为1表示该行块，列块只能有一个堡垒，
建立超级源点S和汇点T，S向每个行块连边，每个列块向T连边

#### C++ 代码

```cpp
#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
#define x first
#define y second
typedef long long LL;
typedef pair<int, int> PII;

const int N = 200, M = 2010, INF = 0x3f3f3f3f;

int n, m, S, T;
int h[N], e[M], ne[M], f[M], idx;
int cur[N], q[N], d[N];
int rid, cid;
char g[N][N];
PII id[N][N];

void add(int a, int b, int c) {
    e[idx] = b, f[idx] = c, ne[idx] = h[a], h[a] = idx++;
    e[idx] = a, f[idx] = 0, ne[idx] = h[b], h[b] = idx++;
}

int find(int u, int limit) {
    if (u == T) return limit;
    int flow = 0;
    for (int i = cur[u]; ~i && flow < limit; i = ne[i]) {
        cur[u] = i;
        int v = e[i];
        if (f[i] && d[v] == d[u] + 1) {
            int t = find(v, min(f[i], limit - flow));
            if (!t) d[v] = -1;
            flow += t, f[i] -= t, f[i ^ 1] += t;
        }
    }
    return flow;
}

bool bfs() {
    memset(d, -1, sizeof d);
    int hh = 0, tt = 0;
    q[0] = S, cur[S] = h[S], d[S] = 0;
    while(hh <= tt) {
        int u = q[hh++];
        for (int i = h[u]; ~i; i = ne[i]) {
            int v = e[i];
            if (f[i] && d[v] == -1) {
                d[v] = d[u] + 1;
                cur[v] = h[v];
                if (v == T) return true;
                q[++tt] = v;
            }
        }
    }
    return false;
}

int dinic() {
    int res = 0, flow;
    while(bfs()) while(flow = find(S, INF)) res += flow;
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    while(cin >> n, n) {
        for (int i = 1; i <= n; i++) cin >> g[i] + 1;
        memset(h, -1, sizeof h);
        idx = rid = cid = 0;
        
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (g[i][j] != 'X') {
                    if (j == 1 || g[i][j - 1] == 'X') rid++;
                    id[i][j].x = rid;
                }
            }
        }
        for (int j = 1; j <= n; j++) { // 注意先遍历列
            for (int i = 1; i <= n; i++) {
                if (g[i][j] != 'X') {
                    if (i == 1 || g[i - 1][j] == 'X') cid++;
                    id[i][j].y = cid;
                }
            }
        }

        S = 0, T = rid + cid + 2;
        for (int i = 1; i <= rid; i++) add(S, i, 1);
        for (int i = 1; i <= cid; i++) add(i + rid, T, 1);
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (g[i][j] == '.')
                    add(id[i][j].x, id[i][j].y + rid, 1);
            }
        }
        cout << dinic() << endl;
    }
    return 0;
}turn 0;
}
```
