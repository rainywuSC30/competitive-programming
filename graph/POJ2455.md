秘密挤奶机 https://www.acwing.com/activity/content/problem/content/2588/




### 题目描述

给定无向图，从每条边长度li，从 1 到 n 最多t条不相交路径，求 满足题目下最大边权的最小值

1≤T≤200
2≤N≤200
1≤P≤40000
1≤Ai,Bi≤N
1≤Li≤106

#### 样例

```
输入：
7 9 2
1 2 2
2 3 5
3 7 5
1 4 1
4 3 1
4 5 7
5 7 1
1 6 3
6 7 3

输出：
5
```

----------

## 算法1
##### 二分 + 最大流

1. 二分边权mid， 大于mid 容量为0， 否则为1， 因此 最大流大于等于t保证了存在不相交的t个路径；
2. 对于无向图，正向c，反向c等于改变流了0，可以分别建立4条边，也可以合并同向边，这样只要建2个边，容量均初始为 c （正 + 反残 = 反 + 正残 = c）
3. 跑最大流

```cpp
#include <bits/stdc++.h>
using namespace std;
#define x first
#define y second
typedef pair<int, int> PII;
typedef long long LL;

const int N = 210, M = 8e4 + 10, INF = 1e9;

int n, m, S, T, tot;
int h[N], e[M], ne[M], f[M], idx;
int cur[N], q[N], d[N];

void add(int a, int b, int c) {
    e[idx] = b, f[idx] = c, ne[idx] = h[a], h[a] = idx++;
    e[idx] = a, f[idx] = c, ne[idx] = h[b], h[b] = idx++;
}

int find(int u, int limit) {
    if (u == T) return limit;
    int flow = 0;
    for (int i = cur[u]; ~i && flow < limit; i = ne[i]) {
        cur[u] = i;
        int v = e[i];
        if (d[v] == d[u] + 1 && f[i]) {
            int t = find(v, min(f[i], limit - flow));
            if (!t) d[v] = -1;
            flow += t, f[i] -= t, f[i ^ 1] += t;
        }
    }
    return flow;
}

bool bfs(){
    memset(d, -1, sizeof d);
    int hh = 0, tt = 0;
    q[hh] = S, d[S] = 0, cur[S] = h[S];
    while(hh <= tt) {
        int u = q[hh++];
        for (int i = h[u]; ~i; i = ne[i]) {
            int v = e[i];
            if (d[v] == -1 && f[i]) {
                d[v] = d[u] + 1;
                cur[v] = h[v];
                if (v == T) return true;
                q[++tt] = v;
            }
        }
    }
    return false;
}

int dinic() {
    int res = 0, flow;
    while(bfs()) while(flow = find(S, INF)) res += flow;
    return res;
}

bool check(int l) {
    int tmp[M];
    memcpy(tmp, f, sizeof f);
    for (int i = 0; i < idx; i ++) {
        if (f[i] > l) f[i] = 0;
        else f[i] = 1;
    }
    bool res = (dinic() >= tot);
    memcpy(f, tmp, sizeof f);
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> m >> tot;
    S = 1, T = n;
    memset(h, -1, sizeof h);
    for (int i = 1; i <= m; i++) {
        int a, b, l;
        cin >> a >> b >> l;
        add(a, b, l);
    }
    int l = 1, r = 1e6;
    while(l <= r) {
        int mid = l + r >> 1;
        if (check(mid)) r = mid - 1;
        else l = mid + 1;
    }
    cout << l << endl;

    return 0;
}
```
