## 算法1
##### 线段树 

维护最大子段和

维护连续子段和，用 tmax, lmax, rmax 来维护

#### C++ 代码

```cpp
// 线段树维护最大子段和
#include<bits/stdc++.h>
using namespace std;

const int N = 50010;

int n, m;
struct Node {
    int l, r, tmax, lmax, rmax;
} tr[N * 4];
int stk[N], top;

void pushup(int u) {
    Node &root = tr[u], left = tr[u << 1], right = tr[u << 1 | 1];
    
    root.lmax = left.lmax;
    if (left.lmax == left.r - left.l + 1) root.lmax += right.lmax;
    root.rmax = right.rmax;
    if (right.rmax == right.r - right.l + 1) root.rmax += left.rmax;
    root.tmax = max(max(left.tmax, right.tmax), left.rmax + right.lmax);
}

void build(int u, int l, int r) {
    if (l == r) {
        tr[u] = {l, r, 1, 1, 1};
        return;
    }
    tr[u] = {l, r, 0, 0, 0};
    int mid = l + r >> 1;
    build(u << 1, l, mid);
    build(u << 1 | 1, mid + 1, r);
    pushup(u);
}

void modify(int u, int x, int d) {
    if (tr[u].l == x && tr[u].r == x) {
        tr[u] = {x, x, d, d, d};
        return;
    }
    int mid = tr[u].l + tr[u].r >> 1;
    if (x <= mid) modify(u << 1, x, d);
    if (x > mid) modify(u << 1 | 1, x, d);
    pushup(u);
}

int query(int u, int x) {
    if (tr[u].l == tr[u].r || tr[u].tmax == 0 || tr[u].tmax == tr[u].r - tr[u].l + 1) {
        return tr[u].tmax;
    }
    int mid = tr[u].l + tr[u].r >> 1;
    if (x <= mid) {
        if (x > mid - tr[u << 1].rmax) return tr[u << 1].rmax + tr[u << 1 | 1].lmax;
        else return query(u << 1, x);
    }
    if (x > mid) {
        if (x <= mid + tr[u << 1 | 1].lmax) return tr[u << 1 | 1].lmax + tr[u << 1].rmax;
        else return query(u << 1 | 1, x);
    }
}

int main() {
    while(cin >> n >> m) {
        top = 0;
        build(1, 1, n);
        char op[2];
        int x;
        while(m--) {
            scanf("%s", op);
            if (op[0] == 'D') {
                scanf("%d", &x);
                modify(1, x, 0);
                stk[++top] = x;
            }
            else if (op[0] == 'Q') {
                scanf("%d", &x);
                cout << query(1, x) << endl;
            }
            else if (op[0] == 'R' && top){
                x = stk[top--];
                modify(1, x, 1);
            }
        }
    }
    return 0;
}
```
