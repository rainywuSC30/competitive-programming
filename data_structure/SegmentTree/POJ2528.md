## 算法1
##### 离散化 + 线段树

维护每个区间的最小值(区间的值为输入的次序，非0) 若最小值存在，则该区间可以露出来避免覆盖两端点，误判区间被覆盖，可以在所有可以成区间的两端点之间插入一个数
注意：
1. 区间修改，区间查询，需要懒标记，lazy向下传递的是占据该节点区间的最近海报索引，表示之前占据该区间的海报被覆盖看不到了
2. 判断是否露出看查询海报区间的序号最小值是否等于该海报索引，是 +1
3. 离散化防止区间丢失，每个区间后面添加一个数
    [1, 10][1,3][5,10] 离散化后仅有4个点 [1,3,5,10], 查询区间返回了[1,3][5,10]最小值，但是丢失了[1,10]区间，
     因此添加r + 1节点 [1,3,4,5,10,11] 查询[1,3,4,5,10] 因此不会丢失[1, 10] 区间信息；

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
#define x first
#define y second
typedef pair<int, int> PII;

const int N = 3e4 + 10;

int n;
PII q[N];
struct Node {
    int l, r, minv, tag;
} tr[N * 4];

vector<int> alls;

int find(int x) {
    int l = 0, r = alls.size() - 1;
    while(l <= r) {
        int mid = l + r >> 1;
        if (alls[mid] >= x) r = mid - 1;
        else l = mid + 1;
    }
    return l + 1; // 注意线段树中的索引从1开始，这里要加1偏移量
}

void pushup(int u) {
    tr[u].minv = min(tr[u << 1].minv, tr[u << 1 | 1].minv);
}

void pushdown(int u) {
    if (tr[u].tag) {
        Node &root = tr[u], &left = tr[u << 1], &right = tr[u << 1 | 1];
        left.minv = right.minv = left.tag = right.tag = root.tag; // 更新子节点tag
        root.tag = 0;
    }
}

void build(int u, int l, int r) {
    if (l == r) {
        tr[u] = {l, r, INT_MAX, 0};
        return;
    }
    tr[u] = {l, r, INT_MAX, 0};
    int mid = l + r >> 1;
    build(u << 1, l, mid);
    build(u << 1 | 1, mid + 1, r);
    pushup(u);
}

void modify(int u, int l, int r, int x) {
    if (tr[u].l >= l && tr[u].r <= r) {
        tr[u].minv = x, tr[u].tag = x;
        return;
    }
    pushdown(u);
    int mid = tr[u].l + tr[u].r >> 1;
    if (l <= mid) modify(u << 1, l, r, x);
    if (r > mid) modify(u << 1 | 1, l, r, x);
    pushup(u);
}

int query(int u, int l, int r) {
    if (tr[u].l >= l && tr[u].r <= r) return tr[u].minv;
    int mid = tr[u].l + tr[u].r >> 1;
    int res = INT_MAX;
    pushdown(u);
    if (l <= mid) res = min(res, query(u << 1, l, r));
    if (r > mid) res = min(res, query(u << 1 | 1, l, r));
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T;
    cin >> T;
    while (T--) {
        alls.clear();
        cin >> n;
        for (int i = 1; i <= n; i++) {
            int a, b;
            cin >> a >> b;
            q[i] = {a, b};
            alls.push_back(a);
            alls.push_back(b);
            alls.push_back(b + 1); // 防止离散化后漏掉区间两个端点被其他不同区间覆盖的情况。
        }
        sort(alls.begin(), alls.end());
        alls.erase(unique(alls.begin(), alls.end()), alls.end());
        build(1, 1, alls.size());
        for (int i = 1; i <= n; i++) {
            auto [l, r] = q[i];
            l = find(l), r = find(r);
            modify(1, l, r, i);
        }
        int res = 0;
        for (int i = 1; i <= n; i++) {
            auto[l, r] = q[i];
            l = find(l), r = find(r);
            int idx = query(1, l, r);
            if (i == idx) res++;
        }
        cout << res << endl;
    }
    return 0;
}
```
