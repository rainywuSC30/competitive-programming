## 算法1
##### 线段树

多个懒标记维护，
区间维护形式：a*mul + add; 加 mul = 1, add += add,  乘： mul = mul * mul, add = mul * add + add;

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 100010, p = 10007;
using LL = long long;

int n, m;

struct SegTree
{
    int l, r;
    int sum1, sum2, sum3, add, mul; 
}tr[N * 4];

void pushup(int u)
{
    tr[u].sum1 = (LL)(tr[u << 1].sum1 + tr[u << 1 | 1].sum1) % p;
    tr[u].sum2 = (LL)(tr[u << 1].sum2 + tr[u << 1 | 1].sum2) % p;
    tr[u].sum3 = (LL)(tr[u << 1].sum3 + tr[u << 1 | 1].sum3) % p;
}

void eval(SegTree &t, int add, int mul) // 推公式
{
    t.sum3 = ((LL)t.sum3 * mul * mul % p * mul % p + 
              (LL)t.sum2 * 3 * mul % p * mul * add % p + 
              (LL)t.sum1 * 3 * mul % p * add * add % p +
              (LL)(t.r - t.l + 1) * add * add % p * add % p) % p;
    t.sum2 = ((LL)t.sum2 * mul % p * mul % p + (LL)t.sum1 * 2 * mul % p * add % p + 
              (LL)(t.r - t.l + 1) * add * add % p) % p;
    t.sum1 = ((LL)t.sum1 * mul % p + (LL)(t.r - t.l + 1) * add % p) % p;

    t.add = ((LL)t.add * mul + add) % p;
    t.mul = (LL)t.mul * mul % p;
}

void pushdown(int u)
{
    eval(tr[u << 1], tr[u].add, tr[u].mul);
    eval(tr[u << 1 | 1], tr[u].add, tr[u].mul);
    tr[u].add = 0;
    tr[u].mul = 1;
}

void build(int u, int l, int r)
{
    if(l == r) tr[u] = {l, r, 0, 0, 0, 0, 1};
    else
    {
        tr[u] = {l, r, 0, 0, 0, 0, 1};
        int mid = l + r >> 1;
        build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
        pushup(u);
    }
}

void modify(int u, int l, int r, int add, int mul)
{
    if(tr[u].l >= l && tr[u].r <= r) eval(tr[u], add, mul);
    else 
    {
        pushdown(u);
        int mid = tr[u].l + tr[u].r >> 1;
        if(l <= mid) modify(u << 1, l, r, add, mul);
        if(r > mid) modify(u << 1 | 1, l, r, add, mul);
        pushup(u);
    }
}

int query(int u, int l, int r, int k)
{
    if(tr[u].l >= l && tr[u].r <= r)
    {
        if(k == 1) return tr[u].sum1;
        else if(k == 2) return tr[u].sum2;
        else return tr[u].sum3;  
    }
    else 
    {
        pushdown(u);
        int mid = tr[u].l + tr[u].r >> 1;
        int sum = 0;
        if(l <= mid) sum = query(u << 1, l, r, k); 
        if(r > mid) sum = (LL)(sum + query(u << 1 | 1, l, r, k)) % p;
        pushup(u);
        return sum % p;
    }
}

int main()
{
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    while(cin >> n >> m && n)
    {
        build(1, 1, n);
        while(m --)
        {
            int op, l, r, c; cin >> op >> l >> r >> c;
            if(op == 1) modify(1, l, r, c, 1);
            else if(op == 2) modify(1, l, r, 0, c);
            else if(op == 3) modify(1, l, r, c, 0);   
            else cout << query(1, l, r, c) << endl;   
        }
    }
    return 0;
}
```
