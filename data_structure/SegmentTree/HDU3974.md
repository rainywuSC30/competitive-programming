分配任务 https://www.acwing.com/problem/content/4349/


### 题目描述

给定一个上下级关系的树，所有下属的任务和上级一样，两个操作，修改一个节点的任务类型，查询一个节点的任务类型

#### 样例

```
输入：
1
5
4 3
3 2
1 3
5 2
5
C 3
T 2 1
C 3
T 3 2
C 3

输出：
Case #1:
-1 
1 
2
```

----------

## 算法1
##### 线段树

从根节点开始dfs, 设点u第一次遍历到的时间戳为dfn[u], 最后一次回溯到的时间戳为dfr[u];
容易发现对于u的任意子节点v, 有dfn[v] > dfn[u], dfr[v] < dfr[u];
且不同子节点的区间[dfn[v], dfr[v]]之间没有交集;

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;

const int N = 5e4 + 10;

int n, m;
int dfn[N], dfr[N], timestamp;
int h[N], e[N], ne[N], idx;
int din[N];
struct Node {
    int l, r, t, lazy;
} tr[N * 4];

void add(int a, int b) {
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

void dfs(int u) {
    dfn[u] = ++timestamp;
    for (int i = h[u]; ~i; i = ne[i]) {
        int v = e[i];
        dfs(v);
    }
    dfr[u] = timestamp;
}

void pushdown(int u) {
    Node &root = tr[u], &left = tr[u << 1], &right = tr[u << 1 | 1];
    if (root.lazy != -1) {
        left.t = right.t = left.lazy = right.lazy = root.lazy;
        root.lazy = -1;
    }
}

void build(int u, int l, int r) {
    if (l == r) {
        tr[u] = {l, r, -1, -1};
        return;
    }
    tr[u] = {l, r, -1, -1};
    int mid = l + r >> 1;
    build(u << 1, l, mid);
    build(u << 1 | 1, mid + 1, r);
}

void modify(int u, int l, int r, int d) {
    if (tr[u].l >= l && tr[u].r <= r) {
        tr[u].t = d, tr[u].lazy = d;
        return;
    }
    pushdown(u);
    int mid = tr[u].l + tr[u].r >> 1;
    if (l <= mid) modify(u << 1, l, r, d);
    if (r > mid) modify(u << 1 | 1, l, r, d);
}

int query(int u, int x) {
    if (tr[u].l == tr[u].r) {
        return tr[u].t;
    }
    pushdown(u);
    int mid = tr[u].l + tr[u].r >> 1;
    if (x <= mid) return query(u << 1, x);
    if (x > mid) return query(u << 1 | 1, x);
}

int main() {
    int T;
    cin >> T;
    int t = 0;
    while(T--) {
        printf("Case #%d:\n", ++t);
        memset(dfn, 0, sizeof dfn);
        memset(din, 0, sizeof din);
        idx = timestamp = 0;
        cin >> n;
        memset(h, -1, sizeof h);
        for (int i = 1; i < n; i++) {
            int a, b;
            scanf("%d%d", &a, &b);
            din[a]++;
            add(b, a);
        }
        for (int i = 1; i <= n; i++) {
            if (!din[i]) dfs(i);
        }
        build(1, 1, timestamp);
        cin >> m;
        while(m--) {
            char op[2];
            int x, y;
            scanf("%s", op);
            if (op[0] == 'T') {
                scanf("%d%d", &x, &y);
                modify(1, dfn[x], dfr[x], y);
            }
            else if (op[0] == 'C') {
                scanf("%d", &x);
                int res = query(1, dfn[x]);
                cout << res << endl;
            }
        }
    }
    return 0;
}
```
