### 题目描述

给定 n 个坐标的 m 个相对关系,求出输入的第 i 条语句时两个坐标的曼哈顿距离。

#### 样例

```
输入：
7 6
1 6 13 E
6 3 9 E
3 5 7 S
4 1 3 N
2 4 20 W
4 7 2 S
3
1 6 1
1 4 3
2 6 6

输出：
13
-1
10
```

----------

## 算法1
##### 带权并查集
曼哈顿距离需要去维护两个方向上的距离，所以这里的d[x].x代表着点x到父节点的x轴方向上的距离，d[x].y代表着点x到父节点的y轴方向上的距离，在计算的时候仍可参考向量计算方式。

#### 参考文献

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
#define x first
#define y second 
typedef pair<int, int> PII;

const int N = 4e4 + 10, M = 1e4 + 10;

int n, m, k;
int p[N];
struct Edge {
    int a, b, l, dir;
} edges[N];
struct Query {
    int a, b, c, idx;
} queries[M];
PII d[N];

int find(int x) {
    if (p[x] != x) {
        int u = find(p[x]);
        d[x].x += d[p[x]].x;
        d[x].y += d[p[x]].y;
        p[x] = u;
    }
    return p[x];
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) p[i] = i;
    for (int i = 1; i <= m; i++) {
        int a, b, l;
        char c;
        cin >> a >> b >> l >> c;
        if (c == 'E') edges[i] = {a, b, l, 0};
        if (c == 'W') edges[i] = {b, a, l, 0};
        if (c == 'N') edges[i] = {b, a, l, 1};
        if (c == 'S') edges[i] = {a, b, l, 1};
    }
    cin >> k;
    for (int i = 1; i <= k; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        queries[i] = {a, b, c, i};
    }
    sort(queries + 1, queries + k + 1, [](Query &a, Query &b) {
        return a.c < b.c; 
    });
    int t = 1, ans[M] = {0};
    for (int i = 1; i <= m; i++) {
        auto [a, b, l, dir] = edges[i];
        int pa = find(a), pb = find(b);
        if (pa != pb) {
            p[pa] = pb;
            if (dir == 0) {
                d[pa].x = l + d[b].x - d[a].x;  // - d[a] + dis + d[b] = x (向量计算)
                d[pa].y = d[b].y - d[a].y;
            }
            else {
                d[pa].x = d[b].x - d[a].x;
                d[pa].y = l + d[b].y - d[a].y;
            }
        }
        while(queries[t].c == i) {
            int a = queries[t].a, b = queries[t].b, idx = queries[t].idx;
            int pa = find(a), pb = find(b);
            if (pa == pb) {
                ans[idx] = abs(d[a].x - d[b].x) + abs(d[a].y - d[b].y);
            }
            else {
                ans[idx] = -1;
            }
            t++;
        }
    }
    for (int i = 1; i <= k; i++) cout << ans[i] << endl;
    return 0;
}
```
