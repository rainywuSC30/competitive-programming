## 算法1
##### 异或前缀和



pre_xor(i) 表示区间 [1, i]的异或前缀和

1. 对于区间 [l,r]的异或值为 0， 表示 pre_xor(r) = pre_xor(l - 1);

   构造一个数组 pre_xor = [0, 1,....,n],令 pre_xor(r) = pre_xor(l - 1);

2. 然后根据公式 pre_xor(i) = a[i] ^ pre_xor(i - 1)反推 a数组即可

#include<bits/stdc++.h>
using namespace std;
#define endl '\n'

typedef long long LL;

const int N = 4e5 + 10, M = 4e5 + 10, INF = 0x3f3f3f3f;

int n, l, r;
int a[N], b[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int T;
    cin >> T;
    while(T--) {
        cin >> n >> l >> r;
        for (int i = 0; i <= n; i++) b[i] = i;
        b[l - 1] = b[r];
        for (int i = 1; i <= n; i++) 
            a[i] = b[i - 1] ^ b[i];
        for (int i = 1; i <= n; i++) cout << a[i] << " ";
        cout << endl;
    }
    return 0;
}
