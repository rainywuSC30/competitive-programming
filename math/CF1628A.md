## 算法1
##### 双指针 + 哈希表

1. 根据题目要求 要选择最短的子数组使得 mex值最大
2. 双指针将数组切段
   当 j指针后的 cnt[mex]值为0的时候，意味着可得到最大 mex值的数组最短，切
3. 双指针 i ，j 只增不降，O(n) 时间； 数组中的每个数字 只会更新一次 mex值，
    因此虽然三重循环，但是整体上 是 O(n)时间复杂度

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
#define x first
#define y second
typedef long long LL;
typedef pair<int, int> PII;

const int N = 2e5 + 10, M = 4e5 + 10, INF = 0x3f3f3f3f;

int n;
int a[N];
int cnt[N];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int T;
    cin >> T;
    while(T--) {
        cin >> n;
        memset(cnt, 0, sizeof cnt);
        for (int i = 1; i <= n; i++) cin >> a[i];

        for (int i = 1; i <= n; i++) {
            cnt[a[i]]++;
        }
        int i = 1;
        vector<int> ans;
        while(i <= n) {
            int j = i;
            int mex = 0;
            vector<int> vis(n + 1, 0);
            while(j <= n) {
                vis[a[j]] = 1;
                cnt[a[j]]--;
                while(vis[mex]) mex++;
                if (cnt[mex] == 0) { // 当数组中不存在mex时，该段最短使得mex最大
                    ans.push_back(mex);
                    j++;
                    break;
                }
                j++;
            }
            i = j;
        }
        cout << ans.size() << endl;
        for (auto x: ans) cout << x << " ";
        cout << endl;
    }
    return 0;
}
```
