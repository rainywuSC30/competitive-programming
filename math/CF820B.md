#include<bits/stdc++.h>
using namespace std;
#define endl '\n'

const int N = 2e5 + 10, M = 4e5 + 10, INF = 0x3f3f3f3f;

double n, a;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> a;
    double d = (n - 2) * 180 / n / (n - 2);
    int l = (int)(a / d);
    if (!l) l = 1;
    else if (l >= n - 2) l = n - 2;
    int ans= l + 2;
    if (l != n - 2 && fabs(a - l * d) > fabs(a - (l + 1) * d))
        ans++;
    cout << "2 1 " << ans << endl;
    return 0;
}
