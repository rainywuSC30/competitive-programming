## 算法1
##### 前缀异或和

1. 题目要求 选出的n个数字任何区间内的 异或和不为0，

2. 构造[l, r]区间内的异或前缀和数组 b

   $b[i] = a[i]$  ^ $b[i - 1]$ ,因此  $a_l$ ^ $a_{l + 1}$^  ... $a_r$  = b[l-1] ^ b[r]                     

3. 因此要求 b[l-1] != b[r], 且不能选择 0

共有 $A_{2^m-1}^n$ 选法

 #include<bits/stdc++.h>
using namespace std;
#define endl '\n'
#define x first
#define y second
#define int long long
typedef pair<int, int> PII;

const int N = 2e5 + 10, M = 4e5 + 10, INF = 0x3f3f3f3f;

int n, m, mod = 1e9 + 9;

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> m;
    int p = 1, ans = 1;
    for (int i = 1; i <= m; i++) 
        p = p * 2 % mod;
    for (int i = 1; i <= n; i++) {
        ans = ans * (p - i) % mod;
    }
    cout << ans << endl;
    return 0;
}
