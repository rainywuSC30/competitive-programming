## 算法1
##### 差分 + 二分 + 后缀数组

1. 因为存在区间整体加减，使用差分保证主题不变
2. 后缀数组得到height 数组和 sa数组，
   判断主题是否重复：根据两个后缀对的等级为（i,j），在height满足[i, j]的lcp为 k（因为差分，4个代表原来数组长度5，abs(sa[i] - sa[j]) > k;
4. height数组上，区间越大，可能的长度越小，但是主题越不可能重复; 二分最大可能长度，转化为判定问题
   给定一个height数组，判断是否存在区间[i, j]使得 lcp(i, j) >= k, 且区间 sa最大值 - sa最小值 > k;

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
const int N = 5010;

int n, m;
int s[N];
int x[N], y[N], c[N], height[N], rk[N], sa[N];
int p[N], smax[N], smin[N];

void get_sa() {
    for (int i = 1; i <= n; i++) c[x[i] = s[i]]++;
    for (int i = 1; i <= m; i++) c[i] += c[i - 1];
    for (int i = n; i; i--) sa[c[x[i]]--] = i;
    
    for (int k = 1; k <= n; k <<= 1) {
        int num = 0;
        for (int i = n - k + 1; i <= n; i++) y[++num] = i;
        for (int i = 1; i <= n; i++) {
            if (sa[i] > k) 
                y[++num] = sa[i] - k;
        }
        for (int i = 1; i <= m; i++) c[i] = 0;
        for (int i = 1; i <= n; i++) c[x[i]]++;
        for (int i = 2; i <= m; i++) c[i] += c[i - 1];
        for (int i = n; i; i--) sa[c[x[y[i]]]--] = y[i], y[i] = 0;
        swap(x, y);
        x[sa[1]] = 1, num = 1;
        for (int i = 2; i <= n; i++) {
            x[sa[i]] = (y[sa[i]] == y[sa[i - 1]] && y[sa[i] + k] == y[sa[i - 1] + k]) ? num : ++num;
        }
        if (num == n) break;
        m = num;
    }
}

void get_height() {
    for (int i = 1; i <= n; i++) rk[sa[i]] = i;
    for (int i = 1, k = 0; i <= n; i++) {
        if (rk[i] == 1) continue;
        if (k) k--;
        int j = sa[rk[i] - 1];
        while(i + k <= n && j + k <= n && s[i + k] == s[j + k]) k++;
        height[rk[i]] = k;
    }
}

bool check(int mid) {
    int sa_min = sa[1], sa_max = sa[1];
    for (int i = 2; i <= n; i++) {
        if (height[i] >= mid) {
            sa_min = min(sa_min, sa[i]);
            sa_max = max(sa_max, sa[i]);
            if (sa_max - sa_min > mid) return true;
        }
        else {
            sa_min = sa_max = sa[i];
        }
    }
    return false;
}

int find(int x) {
    if (p[x] != x) p[x] = find(p[x]);
    return p[x];
}

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> s[i];
    for (int i = n; i; i--) s[i] = s[i] - s[i - 1] + 100;
    m = 200;
    get_sa();
    get_height();
    
    // 1. 二分枚举最大长度
    int l = 0, r = 5010;
    while(l <= r) {
        int mid = l + r >> 1;
        if (check(mid)) l = mid + 1;
        else r = mid - 1;
    }
    if (r >= 4) cout << r + 1 << endl;
    else cout << 0 << endl;
    
    return 0;
}
```
