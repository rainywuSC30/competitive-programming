#include <bits/stdc++.h>
using namespace std;
#define x first
#define y second
#define int long long
typedef pair<int, int> PII;

const int N = 1e6 + 10;

int n, m;
char s[N];
int x[N], y[N], sa[N], height[N], c[N], rk[N];
int stk[N], top;
int R[N], L[N], f[N];

void get_sa() {
    for (int i = 1; i <= n; i++) c[x[i] = s[i]]++;
    for (int i = 2; i <= m; i++) c[i] += c[i - 1];
    for (int i = n; i; i--) sa[c[x[i]]--] = i;
    
    for (int k = 1; k <= n; k <<= 1) {
        int num = 0;
        for (int i = n - k + 1; i <= n; i++) y[++num] = i;
        for (int i = 1; i <= n; i++) {
            if (sa[i] > k) {
                y[++num] = sa[i] - k;
            }
        }
        for (int i = 1; i <= m; i++) c[i] = 0;
        for (int i = 1; i <= n; i++) c[x[i]] ++;
        for (int i = 2; i <= m; i++) c[i] += c[i - 1];
        for (int i = n; i; i--) sa[c[x[y[i]]]--] = y[i], y[i] = 0;
        swap(x, y);
        x[sa[1]] = 1, num = 1;
        for (int i = 2; i <= n; i++) {
            x[sa[i]] = (y[sa[i]] == y[sa[i - 1]] && y[sa[i] + k] == y[sa[i - 1] + k]) ? num : ++num;
        }
        if (num == n) break;
        m = num;
    }
}

void get_height() {
    for (int i = 1; i <= n; i++) rk[sa[i]] = i;
    for (int i = 1, k = 0; i <= n; i++) {
        if (rk[i] == 1) continue;
        if (k) k--;
        int j = sa[rk[i] - 1];
        while(i + k <= n && j + k <= n && s[i + k] == s[j + k]) k++;
        height[rk[i]] = k;
    }
}

signed main() {
    ios::sync_with_stdio (false);
    cin.tie(0);
    cout.tie(0);
    scanf("%s", s + 1);
    n = strlen(s + 1);
    m = 130;
    get_sa();
    get_height();

    // 常数项
    int res = n * (n - 1) * (n + 1) / 2;
    
    // 贡献法:单调栈维护区间最小值
    // 方法1：找到以 height[i]作为做小值区间的左端点和右端点
    stk[++top] = 1;
    // for (int i = 2; i <= n; i++) {
    //     while(top && height[stk[top]] > height[i]) 
    //         R[stk[top--]] = i;
    //     L[i] = stk[top];
    //     stk[++top] = i;
    // }
    // while(top) R[stk[top--]] = n + 1;
    // for (int i = 2; i <= n; i++) {
    //     res -= 2 * (R[i] - i) * (i - L[i]) * height[i];
    // }

    // 方法2：使用动态规划f[i] 递推
    // f[i]表示以i为右端点的区间中，最小值之和
    for (int i = 2; i <= n; i++) {
        while(top && height[stk[top]] > height[i]) top--;
        int j = stk[top];
        f[i] = f[i] + (i - j) * height[i];
        res -= 2 * f[i];
        stk[++top] = i;
    }
    printf("%lld\n", res);
    return 0;
}
