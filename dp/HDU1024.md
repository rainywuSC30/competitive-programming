## 算法1
##### 划分dp + 滚动数组

f[i][j][k] 表示前 i 个数 分成 j 组 第i个数选或不选的最大值

#### 参考文献

#### C++ 代码
```cpp
#include<bits/stdc++.h>
using namespace std;
#define max(a, b) (a > b ? a : b)
typedef long long LL;

const int N = 1e6 + 10, M = 1010;

int n, m;
int s[N];
LL f[2][M][2];

int main() {
    while(~scanf("%d %d", &m, &n)) {
        for (int i = 1; i <= n; i++) {
            scanf("%d", &s[i]);
        }
        memset(f, -0x3f, sizeof f);
        f[0][0][0] = 0;
        for (int i = 1; i <= n; i++) {
            f[i & 1][0][0] = 0;
            int cur = i & 1, pre = (i - 1) & 1;
            for (int j = 1; j <= m; j++) {
                // 选 i 前i个数划分j组，i新组
                f[cur][j][1] = max(f[pre][j - 1][0], f[pre][j - 1][1]) + s[i];
                // 选 i 前i个数划分j组, i旧组
                f[cur][j][1] = max(f[cur][j][1], f[pre][j][1] + s[i]);
                // 不选 i;
                f[cur][j][0] = max(f[pre][j][0], f[pre][j][1]);
            }
        }
        printf("%lld\n", max(f[n & 1][m][0], f[n & 1][m][1]));
    }
}
```
